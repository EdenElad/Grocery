### שיטה 2: שימוש ב-APIs רשמיים
```javascript
// דוגמה - Google Places API
const findStore = async (name, city) => {
  const response = await fetch(
    `https://maps.googleapis.com/maps/api/place/findplacefromtext/json?
input=${name} ${city}&
inputtype=textquery&
key=YOUR_API_KEY`
  );
  return response.json();
};
```
